#include<iostream>
#include<algorithm>
using namespace std;

int main(int argc, char* argv[]){

  int notPrime; // Counter to keep track of how many numbers that can divide evenly into the argument
  int numOfArgs = argc - 1;   // Number of arguments passed in (1 is subtracted since an empty space is always added to the end of an argument)
  string arg = argv[1]; // Retrieve user's argument from the command line and set it as a string
  int argument = atoi(arg.c_str()); // Convert the string to an integer

  // Close the program if there is more than one argment
  if(numOfArgs > 1)
    cout << "Invalid number of arguments. There should only be one argument." << endl;
  else {
    for(int i = 2; i < argument; i++){
      if(argument % i == 0) // If the argument has a remainder of 0 after being divided by the numbers before it, it is not prime.
	notPrime++; // Increase counter for numbers that can multiply into the argument
    }
  
    /* Determine if argument is prime or not. The number 2 is also a parameter in this check because a number is 
       known to be prime if the only numbers it can be divided by evenly are 1 or itself. Because the loop previously seen
       starts at 2 in order to better determine if the argument is prime, it completely skips the number 2, which is what 
       the user could have put in as an argument. */
    if(notPrime == 0 || argument == 2)
      cout << "prime\n" << endl;
    else
      cout << "composite\n" << endl;
  }
  return 0;
}
