//This program determines whether an integer given by command line argument is prime.

#include<iostream>
#include<stdlib.h>

using namespace std;

bool isPrime(int input);

int main(int argc, char** argv) {
	if (argc != 2) { //invalid argument count
		cout << "Invalid number of arguments, " << argc - 1 << " given, 1 expected." << endl;
		return 0;
	}

	string outputString;

	 //This will point to the character in the string directly after the last parsed value. Later we check if it's
	 //the null character, meaning it's the end of the string.
	char* endPointer = NULL;
	int parsedInt = strtol(argv[1], &endPointer, 10);

	if(*endPointer == '\0') { //If the entire inputted string is an integer
		if(isPrime(parsedInt)) {
			outputString = "prime";
		} else {
			outputString = "composite";
		}
	} else { //If the entire inputted string is NOT an integer
		outputString = "composite";
	}

	cout << outputString << endl; //output our result
	return 0;
}


//checks if an integer is a prime number, returns true if it's prime and false if it's composite
bool isPrime(int input) {
	//i begins at 2 because 0 and 1 are factors of everything. i*i<=input is used because any factor greater than the
	//square root of a number will be paired with a factor less than the square root. For example, the factors of 100
	//are 1, 2, 4, 5, 10, 20, 25, 50, 100. The square root of 100 is 10. Note that there are an odd number of factors
	// and that 10 is the centermost factor. Radiating out from the 10 in both directions, the pairs of factors multiply
	// to 100: 5*20, 4*25, 2*50, 1*100. For a number without an integer square root, the number of factors will be even,
	// and the non-integer square root would fall between the central two factors.
	
	//0, 1, and negatives are not prime by definition
	if(input < 2) {
		return false;
	}
	
	
	for(int i = 2; i*i <= input; i++) { 
		if(input % i == 0) { //If input can be divided by i without a remainder
			return false; //We found a factor of input, meaning it is not a prime.
		}
	}
	return true; //We iterated through every possible integer and failed to find a factor, meaning input is prime.
}
