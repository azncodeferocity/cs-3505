/* Definition of the class's add method. 
 * Returns the result of the addition of two huge numbers
 * as a huge_number.
 */
huge_number huge_number::operator+ (const huge_number& rhs)
{
  // Perform the addition of the values
  string left = value;
  string right = rhs.value;
  string result = add(left, right);

  // Wrap the string result of the addition in a huge_number object.
  return huge_number(result);
}

/*
 * Adds two numbers stored in strings, building a string result.
 */
string huge_number::add (string left, string right)
{
  // Build up a string to contain the result.
  
  string result = "";

  // Work right to left.

  int left_pos  = left.length()  - 1;
  int right_pos = right.length() - 1;

  int carry = 0;

  // Loop, adding columns until no more digits remain.

  while (left_pos >= 0 || right_pos >= 0 || carry > 0)
  {
    // Get the digit from each string, or 0 if none.

    int left_digit  = left_pos  >= 0 ? left[left_pos--]   - '0' : 0;
    int right_digit = right_pos >= 0 ? right[right_pos--] - '0' : 0;

    // Calcuate the sum of the digits.

    int sum = left_digit + right_digit + carry;
    carry = sum / 10;
    sum   = sum % 10 + '0';

    // Put the sum into the new string (at the left side)

    result.insert (0, 1, static_cast<char>(sum) );
    
  }

  // Strip out any leading 0's from our result.  (Might not be needed)

  while (result.length() > 0 && result[0] == '0')
    result.erase(0, 1);

  // If the string is empty, it's a 0.  Put one back.

  if (result.length() == 0)
    result = "0";
  
  return result;
}

/* Multiplies two huge_numbers together and returns a huge_number as a result.
 */
huge_number huge_number::operator* (const huge_number& rhs)
{
  // Perform the multiplication of the values
  string left = value;
  string right = rhs.value;
  string result = multiply(left,right);

  // Wrap the result in a huge_number and return
  return huge_number(result);
}

/*
 * Multiplies two numbers stored in strings, building a string result.
 * The algorithm is long multiplication, starting with the most significant
 * digit of the multiplier.
 */
string huge_number:: multiply (string left, string right)
{
  string result = "0";

  int right_pos = 0;

  while (right_pos < right.length())
  {
    result.append("0");  // Multiply the product by 10.

    int right_digit = right[right_pos++] - '0';

    for (int i = 0; i < right_digit; i++)
      result = add(result, left);
  }

  // Strip out any leading 0's from our result.  (Might not be needed)

  while (result.length() > 0 && result[0] == '0')
    result.erase(0, 1);

  // If the string is empty, it's a 0.  Put one back.

  if (result.length() == 0)
    result = "0";

  return result;
}

/* Subtracts two huge_number objects and returns a huge_number as a result.
 * If the result of the subtraction would yield a negative number, a 0 is returned instead.
 */
huge_number huge_number::operator- (const huge_number& rhs)
{
  // Extract the values from the huge_number objects.
  string top = value;  // "top" and "bot" are from the perspective of subtracting by hand on paper
  string bot = rhs.value;

  // Perform the subtraction and return
  string result = subtract(top, bot);
  return huge_number(result);
}

/* Helper method for the subtraction of huge_numbers.
 * Performs the subtraction of two non-negative integers represented as strings
 * and returns the result as a string.
 */
string huge_number::subtract(string top, string bot)
{
  // If the first string is smaller than the second string,
  // the subtraction would yield a negative result. Return 0.
  if(!is_bigger(top, bot))
    {
      return "0";
    }

  // Make an initial check for equality. If the values are equal, return 0.
  if(top.compare(bot) == 0)
    {
      return "0";
    }

  // Perform the subtraction
  // Starting from the right, subtract the bottom digit from the top.
  // If the bottom digit is too large, borrow from the digit to left and decrement the left digit.
  // If the digit to the left is 0 and thus cannot be decremented, change its value to 9 and
  // Repeat the process with the digit on the left until a digit of value has been found.
  // A digit of value is guaranteed to be found because the result of the subtraction will be
  // non-negative at this point.
  // Repeat this process for all digits from right to left.
  // If the bottom value no longer has a digit available, use a 0 for its value.
  string result = "";
  int top_pos = top.length() - 1;
  int bot_pos = bot.length() - 1;

  while(top_pos >= 0)
    {
      int top_digit = top.at(top_pos) - '0'; // convert the char to an int
      int bot_digit;

      if(bot_pos >= 0)
	{
	  bot_digit = bot.at(bot_pos) - '0'; 
	}
      else 
	{
	  bot_digit = 0; // if bot no longer has digits to provide, use a 0
	}

      // Perform the subtraction on the digits
      // If the result would be negative, borrow from the left.
      if(top_digit - bot_digit < 0)
	{
	  // Add 10 to this digit and decrement the digit to the left

	  top_digit = top_digit + 10;
	  int borrow_pos = top_pos - 1;
	  
	  while(top.at(borrow_pos) == '0') // Continue borrowing until a value is found
	    {
	      // Change the 0 to a 9 and continue borrowing left.
	      top[borrow_pos] = '9';
	      borrow_pos--;
	    }

	  int temp = top.at(borrow_pos) - '0';
	  temp--;
	  top[borrow_pos] = static_cast<char> (temp) + '0';
	}

      // Subtract the digits and store the result at the beginning of the string. 
      int sub = top_digit - bot_digit;
      result.insert (0, 1, static_cast<char>(sub + '0'));

      // Loop until all digits have been processed.
      top_pos--;
      bot_pos--;
    }

  // Strip out any leading 0's from our result.  (Might not be needed)

  while (result.length() > 0 && result[0] == '0')
    result.erase(0, 1);

  // If the string is empty, it's a 0.  Put one back.

  if (result.length() == 0)
    result = "0";

  return result; 
}

/* Divides this huge_number's value by the argument's value and returns a huge_number
 * as a result.
 */
huge_number huge_number::operator/ (const huge_number& rhs)
{
  // Perform the division
  string numer = value;
  string denom = rhs.value;
  string result = divide(numer,denom);

  // Wrap the result in a huge_number and return
  return huge_number(result);
}

/* Helper method for the division of huge_numbers.
 * Divides the left argument by the right and 
 * and returns the result as a string.
 *
 * The arguments are string representations of non-negative integers. Ex: "1025".
 */
string huge_number::divide(string numer, string denom)
{
  // Prepare to record the total divisions made
  string total_divisions ="0"; 

  while(is_bigger(numer, denom))
    {
      // Pad the denominator with enough 0's until it is one order of magnitude less than the numerator.
      // Record this padding
      int padding = 0; 
      string padded_denom = denom;
      while(padded_denom.length() < numer.length() - 1)
	{
	  padding++; // record the padding
	  padded_denom.append("0"); // add a 0
	}

      // Repeatedly subtract the padded denominator from the numerator until the subtraction would yield 
      // a negative result. Do not perform this last subtraction.
      // Record these divisions.
      string divisions = "0";
      string last_num = "";

      while(!(numer.compare("0") == 0))
	{
	  last_num = numer; // record the numerator before this
	  divisions = add(divisions, "1"); // record the division
	  numer = subtract(numer, padded_denom); // perform the subtraction
	}

      // The last numerator before reaching 0 is significant.
      numer = last_num;
      divisions = subtract(divisions, "1"); // an extra division will be performed to reach 0, remove it.

      // Add the padding to divisions and record this in the overall result. 
      // Repeat this entire process until the original denominator is larger than the current numerator.
      for(int i = 0; i < padding; i++)
	{
	  divisions.append("0");
	}

      total_divisions = add(total_divisions, divisions);
    }

  if(numer.compare(denom) == 0) // is_bigger will exit the loop 1 iteration early if they are equal
    {
      total_divisions = add(total_divisions, "1");
    }

  return total_divisions;
}

 // Perform the modulus of the values

/* Helper method for the modulus of huge_numbers.
 * Performs the modulus of two non-negative integers represented as strings
 * and returns the result as a string.
 */
string huge_number::mod(string numer, string denom)
{
  while(is_bigger(numer, denom))
    {
      // Pad the denominator with enough 0's until it is one order of magnitude less than the numerator.
      // Record this padding
      int padding = 0; 
      string padded_denom = denom;
      while(padded_denom.length() < numer.length() - 1)
	{
	  padding++; // record the padding
	  padded_denom.append("0"); // add a 0
	}

      // Repeatedly subtract the padded denominator from the numerator until the subtraction would yield 
      // a negative result. Do not perform this last subtraction.
      // Record these divisions.
      string divisions = "0";
      string last_num = "";

      while(!(numer.compare("0") == 0))
	{
	  last_num = numer; // record the numerator before this
	  divisions = add(divisions, "1"); // record the division
	  numer = subtract(numer, padded_denom); // perform the subtraction
	}

      // The last numerator before reaching 0 is significant.
      numer = last_num;
      divisions = subtract(divisions, "1"); // an extra division will be performed to reach 0, remove it.
    }

  // The modulus will be the remaining numerator. 
  return numer;
}
   
/* Helper method to compare two huge_numbers represented as strings.
 * If the left hand side string integer value is larger than the right hand side string integer value,
 * this function returns true. Else, it returns false (even if values are equal).
 *
 * A string integer is a string like "10235". The string integer values must be non-negative. 
 */
bool huge_number::is_bigger (string first, string second)
{
  // If the first string is longer than the second, it must be larger.
  // !** DOES NOT CHECK FOR LEADING 0's, 001 will be reported greater than 20 **!
  if(first.length() > second.length())
    return true;

  // If the second string is longer than the first, the first is smaller. 
  else if(second.length() > first.length())
    return false;

  // At this point, the strings are of equal length. 
  // Starting from the left, determine if the first string has a larger digit. 
  // Return true or false as appropriate if the digit is larger or smaller.
  // If the digits are equal, work right until no more digits remain.
  // If this loop reaches the end of the strings, return false, for the first value is not larger.
  int position = 0;
  while(position < first.length())
    {
      // Retrieve the the characters from each string and convert them to integers
      int first_digit = first.at(position) - '0';
      int second_digit = second.at(position) - '0';
      
      // Compare the integers and return if not equal.
      if(first_digit > second_digit)
	return true;
      else if(second_digit > first_digit)
	return false;
      
      // If the loop was not broken by a return, the values are equal. 
      // Increment position and loop again. 
      position++;
    } 

  // The values must be equal, return false.
  return false;
}
