class huge_number
{

  huge_number(std::string _numString);  // Constructor
  ~huge_number(); // destructor

  std::string get_value();

  huge_number operator+ (const huge_number & rhs) const;
  huge_number operator- (const huge_number & rhs) const;
  huge_number operator* (const huge_number & rhs) const;
  huge_number operator/ (const huge_number & rhs) const;
  huge_number operator% (const huge_number & rhs) const;
  huge_number& operator= (const huge_number & rhs);
  
private:
  std::string numString;
  std::string add(std::string left, std::string right) const;
};
