huge_number huge_number::operator- (const huge_number & rhs) const
{
  // Build up a string to contain the result.
  
  std::string result = "";
  std::string left = numString;
  std::string right = rhs.numString;
  // Work right to left.
  
  int left_length = left.length();
  int right_length = right.length();
  
  if(left_length - right_length > 0)
  {
      right.insert(0, left_length - right_length, '0');
  }
  else if(left_length - right_length < 0)
  {
      left.insert(0, right_length - left_length, '0');
  }

  int left_pos  = left.length()  - 1;
  int right_pos = right.length() - 1;
  int sum = 0;
  int carry = 0;

  // Loop, adding columns until no more digits remain.

  while (left_pos >= 0)
  {
    // Get the digit from each string, or 0 if none.

    int left_digit  = left_pos  >= 0 ? left[left_pos--]   - '0' : 0;
    int right_digit = right_pos >= 0 ? right[right_pos--] - '0' : 0;

    // Calcuate the digits.
    if(left_digit - right_digit - carry >= 0) // if we dont have to borrow
    {
        sum = left_digit - right_digit - carry + '0';
        carry = 0;
    }
    else // if we have to borrow
    {
        sum = 10 + left_digit - right_digit - carry + '0';  // borrow the 10
        carry = 1;											// set carry for next iteration or indicate negative
    }

    // Put the sum into the new string (at the left side)

    result.insert (0, 1, static_cast<char>(sum) );    
  }

  // Strip out any leading 0's from our result.  (Might not be needed)

  while (result.length() > 0 && result[0] == '0')
    result.erase(0, 1);

  // If the string is empty, it's a 0.  Put one back.

  if (result.length() == 0)
    result = "0";
    
  if(carry == 1) // if negative return 0
    result = "0";
  return result;
}

// helper function for add from class example

std::string huge_number::add (std::string left, std::string right) const
{
  // add operation from big_number_example.cpp from class
  // Build up a string to contain the result.
  
  std::string result = "";

  // Work right to left.

  int left_pos  = left.length()  - 1;
  int right_pos = right.length() - 1;

  int carry = 0;

  // Loop, adding columns until no more digits remain.

  while (left_pos >= 0 || right_pos >= 0 || carry > 0)
  {
    // Get the digit from each string, or 0 if none.

    int left_digit  = left_pos  >= 0 ? left[left_pos--]   - '0' : 0;
    int right_digit = right_pos >= 0 ? right[right_pos--] - '0' : 0;

    // Calcuate the sum of the digits.

    int sum = left_digit + right_digit + carry;
    carry = sum / 10;
    sum   = sum % 10 + '0';

    // Put the sum into the new string (at the left side)

    result.insert (0, 1, static_cast<char>(sum) );
    
  }

  // Strip out any leading 0's from our result.  (Might not be needed)

  while (result.length() > 0 && result[0] == '0')
    result.erase(0, 1);

  // If the string is empty, it's a 0.  Put one back.

  if (result.length() == 0)
    result = "0";
  
  return result;
}
