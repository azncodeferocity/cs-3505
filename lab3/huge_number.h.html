#include<string>

/*
 * A huge number class is used to represent non-negative integer values
 * which would exceed the machine's integer capacity. Huge numbers have 
 * capabilities for addition, subtraction, multiplication, division, and
 * modulo.
 */
class huge_number
{
public:
  // Constructor
  // The integer value this class represents  must be provided as a string.
  huge_number(std::string _value); 

  // Copy constructor
  huge_number(const huge_number& other);

  // Destructor
  ~huge_number(); 

  // Getter method
  std::string get_value();

  /* Assignment overload to allow multiple assignments
   * to be carried out in a single statement.
   */
  huge_number& operator= (const huge_number & rhs);

  /* Huge numbers may be added together. 
  * The result of the addition of two huge numbers is a huge number.
  * The + operator will use this method's functionality when called upon two huge_number objects.
  */
  huge_number operator+ (const huge_number& rhs);

 /* Multiplies two huge_number objects together and returns a huge_number as a result.
 */
  huge_number operator* (const huge_number& rhs);

  /* Subtracts two huge_number objects and returns a huge_number as a result.
   * If the result of the subtraction would yield a negative number, a 0 is returned instead.
   */ 
  huge_number operator- (const huge_number& rhs);

  /* Divides two huge_number objects and returns a huge_number as a result.
   * This huge_number object's value is divided by the argument's value. 
   * There is no account for the remainder in this division function. 
   */ 
  huge_number operator/ (const huge_number& rhs);

  /* Performs the modulus of two huge_number objects and returns a huge_number as a result.
   * This huge_number object's value is modded by the argument's value. 
   */ 
  huge_number operator% (const huge_number& rhs);

private:

   /* Helper method for the modulus of huge_numbers.
   * Performs the modulus of two non-negative integers represented as strings
   * and returns the result as a string.
   */
  std:: string mod (std::string numer, std::string denom);

  /* Helper method for the addition of huge_numbers.
   * Performs the addition of two non-negative integers represented as strings
   * and returns the result as a string.
   */
  std:: string add (std::string left, std::string right);

  /* Helper method for the multiplication of huge_numbers.
   * Performs the addition of two non-negative integers represented as strings
   * and returns the result as a string.
   */
  std:: string multiply(std::string left, std::string right);

  /* Helper method for the division of huge_numbers.
   * Divides the left argument by the right and 
   * and returns the result as a string.
   *
   * The arguments are string representations of non-negative integers. Ex: "1025".
   */
  std:: string divide(std::string left, std::string right);

  /* Helper method for the subtraction of huge_numbers.
   * Performs the subtraction of two non-negative integers represented as strings
   * and returns the result as a string.
   */
  std:: string subtract(std::string top, std::string bot);

  /* Helper method to compare two huge_numbers represented as strings.
   * If the left hand side string integer value is larger than the right hand side string integer value,
   * this function returns true. Else, it returns false.
   *
   * A string integer is a string like "10235". The string integer values must be non-negative. 
   */
  bool is_bigger(std::string first, std::string second);
  // String representation of the huge number.
  std::string value;

};
